// shifts chord up N semitones
function transposeChord(chord, n){
  const steps = ["C","C#","D","D#","E","F","F#","G","G#","A","A#","B"];
  const i = steps.findIndex(x => chord.toUpperCase().startsWith(x));
  if(i < 0) return chord;
  const next = steps[(i + n + steps.length) % steps.length];
  return next + chord.slice(steps[i].length);
}

function transposeAll(n){
  document.querySelectorAll('[data-chord]').forEach(el => {
    el.textContent = transposeChord(el.textContent, n);
  });
}

FAQ

Is the scraper reliable?

Yes. It uses a headless browser (Puppeteer) and waits for the DOM. If a site changes markup, Cheerio mappings can be updated without touching the UI.

Can I add local songs without scraping?

Yes. The admin panel supports manual song creation. Chords and lyrics are stored separately in MySQL to keep transpose accurate.

Does dark mode only change background/text colors?

Correct. This page uses a simple theme switch that only flips --bg and --fg tokens, consistent with your preference.
